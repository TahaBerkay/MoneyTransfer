mvn install:install-file -Dfile=ojdbc6.jar -DgroupId=com.oracle -DartifactId=ojdbc6 -Dversion=11.2.0.4 -Dpackaging=jar

CREATE TABLESPACE INGENICO
  DATAFILE 'INGENICO_TS.dat'
    SIZE 20M
    AUTOEXTEND ON;

CREATE USER INGENICO
  IDENTIFIED BY INGENICO
  DEFAULT TABLESPACE INGENICO
  QUOTA UNLIMITED on INGENICO;

GRANT create session TO INGENICO;
GRANT create table TO INGENICO;
GRANT create view TO INGENICO;
GRANT create any trigger TO INGENICO;
GRANT create any procedure TO INGENICO;
GRANT create sequence TO INGENICO;
GRANT create synonym TO INGENICO;
GRANT unlimited tablespace to INGENICO;

CREATE TABLE INGENICO.ACCOUNT
(	
"ID" VARCHAR2(64) NOT NULL, 
"ACCOUNT_NAME" VARCHAR2(256),
"BALANCE" NUMBER(22,2), 
CONSTRAINT "ACCOUNT_PK" PRIMARY KEY ("ID"),
CONSTRAINT "UNIQUE_ACCOUNT" UNIQUE ("ACCOUNT_NAME")
);
